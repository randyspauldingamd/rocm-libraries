set(GFX_SOURCES
    GpuArchManager.cpp
    InstDefDSL.cpp
)

# Add all STINKYTOFU_ARCHS cpp files to GFX_SOURCES
foreach(arch ${STINKYTOFU_ARCHS_TO_BUILD})
    list(APPEND GFX_SOURCES "${arch}.cpp")
endforeach()

list(TRANSFORM GFX_SOURCES PREPEND "${CMAKE_CURRENT_SOURCE_DIR}/src/gfx/")

add_library(gfxisa STATIC
    ${GFX_SOURCES}
)

target_include_directories(gfxisa PUBLIC
    ${CMAKE_CURRENT_SOURCE_DIR}/include
    ${CMAKE_CURRENT_SOURCE_DIR}/../include
    $<BUILD_INTERFACE:${PROJECT_BINARY_DIR}/include> # For generated files
)

target_link_libraries(gfxisa PRIVATE yaml-cpp)

# gfxisa uses yaml-cpp which requires exceptions
# Also needs PIC for linking into Python module
target_compile_options(gfxisa PRIVATE -fexceptions -fPIC)

# Note: gfxisa is only needed by tablegen.
# stinkytofu depends on tablegen_generated target which are generated by tablegen.

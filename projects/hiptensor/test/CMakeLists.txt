if(HIPTENSOR_USE_SYSTEM_GOOGLETEST)
    find_package(GTest REQUIRED)
else()
    include(FetchContent)

    FetchContent_Declare(
        googletest
        GIT_REPOSITORY https://github.com/google/googletest.git
        GIT_TAG v1.17.0
    )
    FetchContent_GetProperties(googletest)
    if(NOT googletest_POPULATED)
        # Fetch the content using default details
        FetchContent_Populate(googletest)
        # Save the shared libs setting, then force to static libs
        set(BUILD_SHARED_LIBS_OLD ${BUILD_SHARED_LIBS})
        set(BUILD_SHARED_LIBS OFF CACHE INTERNAL "Build SHARED libraries" FORCE)
        # Turn off gtest installation
        set(INSTALL_GTEST OFF)
        # On MSVC-style toolchains gtest defaults to the static CRT (/MT), but the
        # rest of hipTensor compiles against the dynamic CRT (/MD). Force gtest to
        # use the dynamic CRT so lld-link does not reject a RuntimeLibrary mismatch.
        if(WIN32)
            set(gtest_force_shared_crt ON CACHE BOOL "Use shared (DLL) run-time lib for gtest" FORCE)
        endif()
        # Add gtest targets as static libs
        add_subdirectory(${googletest_SOURCE_DIR} ${googletest_BINARY_DIR})
        # Restore shared libs setting
        set(BUILD_SHARED_LIBS ${BUILD_SHARED_LIBS_OLD} CACHE INTERNAL "Build SHARED libraries" FORCE)
    endif()
endif()

# On Windows, DLL search order is: application dir -> System32 -> PATH.
# A system-installed ROCm places amdhip64_7.dll and amd_comgr_3.dll in
# System32, which always wins over PATH. If the test binary is compiled with
# a newer toolchain but an older runtime is loaded from System32, kernel launches
# may crash (SEH 0xC0000005) if the code-object format is incompatible.
if(WIN32 AND CREATE_TEST_APP_LOCAL_DEPLOY)
    # hip_DIR is set by find_package(HIP) to <HIP_ROOT>/lib/cmake/hip.
    # Walk up three directory levels to reach <HIP_ROOT>, then append /bin.
    get_filename_component(_hip_cmake_dir  "${hip_DIR}"          DIRECTORY) # .../lib/cmake
    get_filename_component(_hip_lib_dir    "${_hip_cmake_dir}"   DIRECTORY) # .../lib
    get_filename_component(_hip_root       "${_hip_lib_dir}"     DIRECTORY) # <HIP_ROOT>
    set(_hip_bin_dir "${_hip_root}/bin")

    message(STATUS "HIP runtime DLL source: ${_hip_bin_dir}")
    message(STATUS "HIP runtime DLL destination: ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")

    if(EXISTS "${_hip_bin_dir}")
        file(MAKE_DIRECTORY "${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
        file(GLOB _hip_dlls
            "${_hip_bin_dir}/amd_comgr*.dll"   # code-object manager
            "${_hip_bin_dir}/amdhip64*.dll"    # HIP runtime
            "${_hip_bin_dir}/amdhiprtc*.dll"   # HIP RTC
            "${_hip_bin_dir}/rocm_kpack*.dll"  # kernel packaging
        )
        if(_hip_dlls)
            foreach(_dll ${_hip_dlls})
                file(COPY "${_dll}" DESTINATION "${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
            endforeach()
            list(LENGTH _hip_dlls _hip_dll_count)
            message(STATUS "Staged ${_hip_dll_count} HIP runtime DLL(s) into ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
        else()
            message(WARNING "No DLLs found in ${_hip_bin_dir} - tests may load wrong HIP runtime from System32")
        endif()
    else()
        message(WARNING "HIP DLL source directory not found: ${_hip_bin_dir} - "
                        "tests may load wrong HIP runtime from System32")
    endif()
endif()

# Create test executables and deploy
function(add_hiptensor_test BINARY_NAME YAML_CONFIG_FILE TEST_SOURCES)

    # Make sure that all sources are appended to the list.
    list(APPEND TEST_SOURCES ${ARGN})

    message(STATUS "adding hiptensor test: ${BINARY_NAME}")
    message(STATUS "test config file: ${YAML_CONFIG_FILE}")

    # Read in the YAML config file that we wish to bundle
    file(READ ${YAML_CONFIG_FILE} hiptensor_test_YAML_STRING)

    # Build a file header that includes the YAML config string
    # Must define in cmake:
    # - hiptensor_test_NAME
    # - hiptensor_test_YAML_STRING
    # - hiptensor_test_YAML_INCLUDE_FILE
    get_filename_component(hiptensor_test_YAML_DIR "${YAML_CONFIG_FILE}" PATH)
    get_filename_component(hiptensor_test_NAME "${YAML_CONFIG_FILE}" NAME_WE)
    set(hiptensor_test_YAML_INCLUDE_FILE "${hiptensor_test_NAME}.hpp")
    configure_file("${PROJECT_SOURCE_DIR}/test/hiptensor-test-yaml.hpp.in"
                  "${hiptensor_test_YAML_DIR}/${hiptensor_test_YAML_INCLUDE_FILE}")

    # message(STATUS "test config dir: ${hiptensor_test_YAML_DIR}")
    # message(STATUS "test name: ${hiptensor_test_NAME}")
    # message(STATUS "test yaml config: ${hiptensor_test_YAML_STRING}")
    # message(STATUS "test config include: ${hiptensor_test_YAML_INCLUDE_FILE}")

    add_executable(${BINARY_NAME} ${TEST_SOURCES})
    target_compile_options(${BINARY_NAME} PRIVATE ${CLANG_DRIVER_MODE})
    target_link_options(${BINARY_NAME} PRIVATE ${CLANG_DRIVER_MODE})

    # Test must propagate the build interface includes to make sure
    # hiptensor includes are captured at runtime.
    if(WIN32)
        target_link_libraries(${BINARY_NAME} PRIVATE hiptensor::hiptensor hiptensor_llvm GTest::gtest "-L${HIPTENSOR_CLANG_ROOT}/lib")
    else()
        target_link_libraries(${BINARY_NAME} PRIVATE hiptensor::hiptensor hiptensor_llvm GTest::gtest "-L${HIPTENSOR_CLANG_ROOT}/lib"
                              "-Wl,-rpath=$ORIGIN/../${CMAKE_INSTALL_LIBDIR}" "-fno-rtlib-add-rpath")
    endif()

    target_include_directories(${BINARY_NAME} PRIVATE
                               ${CMAKE_CURRENT_SOURCE_DIR}
                               ${composable_kernel_INCLUDES}
                               ${PROJECT_SOURCE_DIR}/library/include
                               ${PROJECT_SOURCE_DIR}/library/src/include
                               ${PROJECT_SOURCE_DIR}/library/src
                               ${PROJECT_SOURCE_DIR}/test
                               ${PROJECT_BINARY_DIR}/include/hiptensor
                               ${PROJECT_BINARY_DIR}/include
                               ${hiptensor_test_YAML_DIR})

    # -Wno-ignored-attributes to avoid warning: __declspec attribute 'dllimport' is not supported [-Wignored-attributes] which is used by msvc compiler
    if(WIN32 AND CMAKE_CXX_COMPILER_ID MATCHES "Clang")
        target_compile_options(${BINARY_NAME} PRIVATE -Wno-ignored-attributes)
    endif()

    # C++ code handles the following symbols defined:
    # - HIPTENSOR_TEST_YAML_INCLUDE
    # When defined, the test code will attempt to load bundled YAML string.
    target_compile_definitions(${BINARY_NAME} PRIVATE -D HIPTENSOR_TEST_YAML_INCLUDE=\"${hiptensor_test_YAML_INCLUDE_FILE}\")

    # Build this test under custom target
    add_dependencies(hiptensor_tests ${BINARY_NAME})

    # Register with ctest
    add_test(NAME ${BINARY_NAME} COMMAND ${BINARY_NAME})
    set_property(TEST ${BINARY_NAME} PROPERTY SKIP_REGULAR_EXPRESSION "HIPTENSOR_STATUS_ARCH_MISMATCH" "unsupported host device")
    # Set working directory to bin/ for proper DLL/resource loading on Windows
    set_property(TEST ${BINARY_NAME} PROPERTY WORKING_DIRECTORY ${CMAKE_RUNTIME_OUTPUT_DIRECTORY})

    # On Windows, ensure the ROCm/HIP DLLs (amdhip64_7.dll, etc.) are discoverable by ctest.
    # ctest may not inherit the user's PATH that includes the ROCm bin directory, causing
    # HIP runtime DLL load failures (SEH 0xC0000005) on the first HIP call in a test fixture.
    if(WIN32)
        set_property(TEST ${BINARY_NAME} PROPERTY ENVIRONMENT
            "PATH=${CMAKE_INSTALL_PREFIX}/bin$<SEMICOLON>${CMAKE_RUNTIME_OUTPUT_DIRECTORY}$<SEMICOLON>$ENV{PATH}")
    endif()

    # Install with rocm pkg
    rocm_install_targets(
    TARGETS ${BINARY_NAME}
    COMPONENT tests
    )

    # Add to testing manifest
    get_target_property(EXE_NAME ${BINARY_NAME} RUNTIME_OUTPUT_NAME)
    if(EXE_NAME STREQUAL "EXE_NAME-NOTFOUND")
        get_target_property(EXE_NAME ${BINARY_NAME} OUTPUT_NAME)
        if(EXE_NAME STREQUAL "EXE_NAME-NOTFOUND")
            set(EXE_NAME "${BINARY_NAME}")
        endif()
    endif()
    file(APPEND "${INSTALL_TEST_FILE}" "add_test(${BINARY_NAME} \"../${BINARY_NAME}\")\n")
    file(APPEND "${INSTALL_TEST_FILE}" "set_tests_properties(${BINARY_NAME} PROPERTIES SKIP_REGULAR_EXPRESSION \"HIPTENSOR_STATUS_ARCH_MISMATCH;unsupported host device\")\n")

    file(APPEND "${INSTALL_SMOKE_TEST_FILE}" "add_test(\"${BINARY_NAME} smoke\" \"../../${BINARY_NAME}\")\n")
    file(APPEND "${INSTALL_SMOKE_TEST_FILE}" "set_tests_properties(\"${BINARY_NAME} smoke\" PROPERTIES SKIP_REGULAR_EXPRESSION \"HIPTENSOR_STATUS_ARCH_MISMATCH;unsupported host device\")\n")

    file(APPEND "${INSTALL_REGRESSION_TEST_FILE}" "add_test(\"${BINARY_NAME} regression\" \"../../${BINARY_NAME}\")\n")
    file(APPEND "${INSTALL_REGRESSION_TEST_FILE}" "set_tests_properties(\"${BINARY_NAME} regression\" PROPERTIES SKIP_REGULAR_EXPRESSION \"HIPTENSOR_STATUS_ARCH_MISMATCH;unsupported host device\")\n")
endfunction()
